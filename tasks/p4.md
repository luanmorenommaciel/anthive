# p4 — `swarm watch` + observability

> **Why this earns its keep:** this is the "where's my money going" command. Live dashboard + Langfuse wiring turn a pile of tmux panes into a fleet you can actually manage.

---

## Goal

1. **`swarm watch`** — live Rich dashboard showing all active sessions with cost/tokens/status, refresh every 3s
2. **Observability wiring** — Claude Code's native OTEL exporter pointed at self-hosted Langfuse; agentswarm emits one parent lifecycle span per session
3. **Cost/token data source** — Langfuse query API (not parsed from logs); dashboard reads live values
4. **Docker compose for Langfuse** — one-command self-host so anyone can run the stack

Key principle: **agentswarm does no custom cost accounting.** Claude Code emits tokens + USD cost via OTEL. Langfuse ingests. agentswarm queries Langfuse. That's it.

## Implementation sketch

### Observability module

```python
# agentswarm/observability.py

import os
from contextlib import contextmanager
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.http.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.resources import Resource

_tracer = None

def init_tracing(service_name: str = "agentswarm", endpoint: str | None = None):
    """Initialize OTEL → Langfuse. Call once at startup."""
    global _tracer
    endpoint = endpoint or os.getenv("OTEL_EXPORTER_OTLP_ENDPOINT",
                                      "http://localhost:3000/api/public/otel")
    resource = Resource.create({"service.name": service_name})
    provider = TracerProvider(resource=resource)
    provider.add_span_processor(BatchSpanProcessor(OTLPSpanExporter(endpoint=endpoint)))
    trace.set_tracer_provider(provider)
    _tracer = trace.get_tracer(service_name)
    return _tracer


@contextmanager
def session_span(session_id: str, task_id: str, agent: str, mode: str):
    """Parent span wrapping a full session lifecycle. Claude Code spans nest under this."""
    tracer = _tracer or init_tracing()
    with tracer.start_as_current_span(
        "swarm.session",
        attributes={
            "swarm.session_id": session_id,
            "swarm.task_id": task_id,
            "swarm.agent": agent,
            "swarm.mode": mode,
        }
    ) as span:
        yield span


def emit_lifecycle_event(session_id: str, from_state: str, to_state: str, note: str = ""):
    """Record a state transition as an OTEL event on the current span."""
    span = trace.get_current_span()
    span.add_event(
        "lifecycle_transition",
        attributes={
            "swarm.session_id": session_id,
            "from": from_state,
            "to": to_state,
            "note": note,
        }
    )
```

### Langfuse query client

```python
# agentswarm/langfuse_client.py

import httpx
from typing import Optional

class LangfuseClient:
    """Read-only client for querying cost/tokens from self-hosted Langfuse."""

    def __init__(self, base_url: str, public_key: str, secret_key: str):
        self.base = base_url.rstrip("/")
        self.auth = (public_key, secret_key)

    def get_session_metrics(self, session_id: str) -> dict:
        """Return {tokens_in, tokens_out, cost_usd, duration_s, trace_id, url}."""
        # Langfuse sessions API: /api/public/sessions?filter=metadata.session_id=<sid>
        resp = httpx.get(
            f"{self.base}/api/public/sessions",
            params={"filter[metadata][session_id]": session_id},
            auth=self.auth,
            timeout=5.0,
        )
        resp.raise_for_status()
        data = resp.json()
        if not data["data"]:
            return {"tokens_in": 0, "tokens_out": 0, "cost_usd": 0.0, "duration_s": 0, "url": None}
        s = data["data"][0]
        return {
            "tokens_in": s.get("usage", {}).get("promptTokens", 0),
            "tokens_out": s.get("usage", {}).get("completionTokens", 0),
            "cost_usd": s.get("totalCost", 0.0),
            "duration_s": s.get("latency", 0) / 1000,
            "trace_id": s.get("id"),
            "url": f"{self.base}/sessions/{s['id']}",
        }
```

### Monitor loop + Rich live dashboard

```python
# agentswarm/monitor.py

import time
from datetime import datetime
from pathlib import Path
from rich.live import Live
from rich.table import Table
from rich.panel import Panel
from rich.spinner import Spinner
from rich.console import Group
from .state import parse_session_log
from .langfuse_client import LangfuseClient

def render_fleet_dashboard(sessions_dir: Path, lf: LangfuseClient, only: list[str] | None = None) -> Panel:
    """Build the live dashboard panel."""
    table = Table(title=f"agentswarm watch · {datetime.now():%H:%M:%S}",
                  header_style="bold cyan")
    table.add_column("ID", style="cyan", no_wrap=True)
    table.add_column("Task")
    table.add_column("Model", style="dim")
    table.add_column("State", justify="center")
    table.add_column("Tokens", justify="right")
    table.add_column("Cost", justify="right")
    table.add_column("Duration", justify="right")
    table.add_column("Last note", style="dim")

    total_cost = 0.0
    n_active = 0

    for log_file in sorted(sessions_dir.glob("*.md")):
        if log_file.name == "_template.md":
            continue
        fm = parse_session_log(log_file)
        if only and fm.session_id not in only:
            continue
        if fm.status == "MERGED":
            continue

        n_active += 1
        metrics = lf.get_session_metrics(fm.session_id)
        total_cost += metrics["cost_usd"]

        state_cell = {
            "INIT": "[yellow]INIT[/]",
            "COOKING": Spinner("dots", text=" COOKING", style="green"),
            "CHECKPOINT": "[cyan]CHECKPOINT[/]",
            "READY-TO-MERGE": "[bold green]READY[/]",
            "BLOCKED": "[red]BLOCKED[/]",
        }.get(fm.status, fm.status)

        tokens = metrics["tokens_in"] + metrics["tokens_out"]
        cost = f"${metrics['cost_usd']:.2f}"
        duration = f"{metrics['duration_s']/60:.1f}m"

        table.add_row(
            fm.session_id,
            fm.task_id[:30],
            fm.model,
            state_cell,
            f"{tokens:,}",
            cost,
            duration,
            fm.last_note[:50],
        )

    footer = f"\n[bold]{n_active} active session(s) · total spend: [green]${total_cost:.2f}[/][/]"
    return Panel(Group(table, footer), title="[bold cyan]agentswarm[/]")


def watch(sessions_dir: Path, lf: LangfuseClient, only: list[str] | None = None,
          budget_alert: float | None = None, refresh: int = 3):
    """Run live dashboard until Ctrl-C."""
    with Live(render_fleet_dashboard(sessions_dir, lf, only), refresh_per_second=1, screen=True) as live:
        while True:
            try:
                time.sleep(refresh)
                live.update(render_fleet_dashboard(sessions_dir, lf, only))
                if budget_alert:
                    check_budget_alert(sessions_dir, lf, budget_alert)
            except KeyboardInterrupt:
                break
```
