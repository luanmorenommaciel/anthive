# p2 — `swarm compose`

> **Why this is the hardest correctness problem:** a wrong prompt wastes a session. Composer generates the prompt deterministically from (task spec + chosen agent + repo context) — no LLM in this step, just templating + lookup.

---

## Goal

Given a task ID, produce:
1. The specialist agent to use (from `.claude/agents/**/*.md`)
2. The full prompt to paste into Claude Code (or send to Managed Agents)
3. The touches-paths and do-not-touch lists
4. The heartbeat + exit-check rules

Output: a frozen `prompts/<task-id>.md` file that `swarm dispatch` consumes.

Key design decision: **no LLM in compose step.** Deterministic. A given task always produces the same prompt unless the task doc or agent definition changes. This is critical for reproducibility.

---

## Success criteria

- [ ] `swarm compose T-20260424-card-violator-fix` produces `prompts/T-20260424-card-violator-fix.md`
- [ ] The generated prompt is equivalent (in shape) to today's hand-authored s2 block from `logs/PROMPTS.md`
- [ ] Agent selection is deterministic: reads `agent:` field from task frontmatter, falls back to heuristic match if missing
- [ ] Prompt includes all non-negotiables from today's PROMPTS.md:
  - session goal (from task body)
  - agent directive (use the Agent tool with `subagent_type=<agent>`)
  - context files to read first (derived from `touches_paths` + dependencies)
  - do-not-touch list (inverted from `touches_paths` across all other in-progress tasks)
  - heartbeat rule
  - exit rule
  - "fill in the session log yourself, don't ask" directive
- [ ] `swarm compose --all-ready` composes prompts for every task in ready list
- [ ] `swarm compose --dry-run <id>` prints the prompt to stdout without writing
- [ ] Pytest: 8+ tests covering agent picker, template rendering, path-conflict do-not-touch

```python
# agentswarm/composer.py

from pathlib import Path
from .schemas import TaskFrontmatter, ReadyList

AGENT_DIR = Path(".claude/agents")

def find_agent(agent_name: str, repo_root: Path) -> Path | None:
    """Find .claude/agents/**/<agent_name>.md (first match)."""
    for candidate in (repo_root / AGENT_DIR).rglob(f"{agent_name}.md"):
        return candidate
    return None

def compose(task: TaskFrontmatter, repo_root: Path, other_ready: list[TaskFrontmatter]) -> str:
    """Generate the full session prompt from task + agent + context."""

    # 1. Resolve agent
    agent_path = find_agent(task.agent, repo_root)
    if not agent_path:
        raise ValueError(f"agent {task.agent} not found under .claude/agents/")

    agent_def = agent_path.read_text()  # frontmatter + body

    # 2. Read task body (the spec after frontmatter)
    task_body = read_task_body(task.path)

    # 3. Compute do-not-touch paths (union of other_ready touches_paths)
    do_not_touch = set()
    for other in other_ready:
        if other.id == task.id:
            continue
        do_not_touch.update(other.touches_paths)

    # 4. Render template
    return PROMPT_TEMPLATE.format(
        session_id=f"s{next_session_id()}-{task.slug}",
        task_id=task.id,
        worktree=f"worktrees/{task.slug}",
        branch=f"session/{task.slug}",
        task_body=task_body,
        agent_name=task.agent,
        agent_path=agent_path,
        allowed_paths="\n".join(f"  - {p}" for p in task.touches_paths),
        forbidden_paths="\n".join(f"  - {p}" for p in sorted(do_not_touch)),
        heartbeat_cmd=f"../../scripts/session_heartbeat.sh {task.slug}",
        exit_check=task.exit_check or "success criteria from task doc",
    )


PROMPT_TEMPLATE = """\
You are running session `{session_id}` from a git worktree at `{worktree}`
on branch `{branch}`. Task ID: `{task_id}`.

═══════════════ GOAL ═══════════════
{task_body}

═══════════════ PRIMARY AGENT ═══════════════
Use the `{agent_name}` specialist agent via the Agent tool (subagent_type={agent_name}).
Agent definition at: {agent_path}

═══════════════ ALLOWED SURFACE ═══════════════
{allowed_paths}

═══════════════ DO NOT TOUCH ═══════════════
{forbidden_paths}

═══════════════ HEARTBEAT RULE ═══════════════
- Every meaningful state change, run:
    {heartbeat_cmd} COOKING "<one-line note>"
- Status transitions: INIT → COOKING → CHECKPOINT → READY-TO-MERGE

═══════════════ EXIT RULE ═══════════════
{exit_check}

When done, run: ../../scripts/session_finish.sh {task_id}
Then push branch and open PR.

═══════════════ AUTONOMY DIRECTIVE ═══════════════
Fill the Goal / Success criteria / touches_paths fields in the session log
yourself from this prompt. Commit that as your first commit:
  chore({task_id}): fill session log from task doc
Do not pause to ask — proceed with the implementation. Only pause for a
human on genuine blockers (missing credentials, real ambiguity).

Start now.
"""
```

### CLI

```python
# agentswarm/cli.py (append)

@app.command()
def compose(
    task_id: str = typer.Argument(None, help="Task ID, or use --all-ready"),
    all_ready: bool = typer.Option(False, "--all-ready"),
    dry_run: bool = typer.Option(False, "--dry-run", help="Print, don't write"),
):
    """Generate the prompt(s) for a task."""
    from .scanner import scan as do_scan
    from .composer import compose as do_compose

    result = do_scan(Path.cwd())

    if all_ready:
        targets = result.ready
    elif task_id:
        targets = [t for t in result.ready if t.id == task_id]
        if not targets:
            typer.echo(f"[red]Task {task_id} not in ready set[/]", err=True)
            raise typer.Exit(1)
    else:
        typer.echo("[red]Must specify task_id or --all-ready[/]", err=True)
        raise typer.Exit(2)

    prompts_dir = Path("prompts")
    prompts_dir.mkdir(exist_ok=True)

    for t in targets:
        other_ready = [o for o in result.ready if o.id != t.id]
        prompt_text = do_compose(t, Path.cwd(), other_ready)
        out = prompts_dir / f"{t.id}.md"
        if dry_run:
            console.print(f"\n─── {out} ───\n")
            console.print(prompt_text)
        else:
            out.write_text(prompt_text)
            console.print(f"✓ [green]{out}[/]")
```
