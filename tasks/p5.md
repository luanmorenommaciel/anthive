# p5 — `swarm merge`

> **Why last in M1:** by the time sessions are dispatching and watched, PR reconciliation is the last link. This replaces today's hand-run "merge session" with an automated dependency-aware reconciler.

---

## Goal

Implement `swarm merge`: reads `logs/merge-queue.md`, lands PRs on `main` in dependency order, runs per-PR exit checks, archives completed session logs.

Protocol:
1. Parse `merge-queue.md` `## Open` section
2. For each unmerged row, resolve dependencies (depends-on: sessions must be already merged)
3. Pick the next mergeable PR (leftmost with all deps satisfied)
4. `git checkout main && git pull && git merge --no-ff session/<slug>`
5. Run the exit_check command (if specified)
6. On success: push main, tick the checkbox, archive session log to `logs/archive/<date>/`
7. On failure: leave merged, open a follow-up task in `tasks/backlog.md`, archive anyway
8. Repeat until queue empty

---

## Success criteria

- [ ] `swarm merge` on today's live queue lands PR #10 (s4) and PR #11 (s3) in correct order
- [ ] Dependency resolution: won't merge B if B depends on A and A is not-yet-merged
- [ ] Conflict detection: if `git merge` fails, pause, print diff, prompt human
- [ ] `--dry-run` shows what would merge, doesn't execute
- [ ] `--auto` no-confirm mode (for cron/scripting)
- [ ] Archive flow: moves session log to `logs/archive/<YYYY-MM-DD>/<slug>.md`, updates `merge-queue.md` with ✅
- [ ] Exit check runs in the worktree BEFORE merge (safer than post-merge)
- [ ] Records merge decision in `logs/decisions/merge-<timestamp>.md` (honesty trail)
- [ ] `swarm merge --reorder` launches interactive reorder UI (questionary) if human wants to override
- [ ] Pytest: 6+ tests

## Implementation sketch

```python
# agentswarm/merger.py

import subprocess
from pathlib import Path
from datetime import datetime
from .schemas import parse_merge_queue, MergeQueueRow

def reconcile(repo_root: Path, dry_run: bool = False, auto: bool = False) -> list[dict]:
    """Iterate merge-queue, land PRs in dependency order."""
    queue_path = repo_root / "logs" / "merge-queue.md"
    rows = parse_merge_queue(queue_path)
    unmerged = [r for r in rows if not r.merged]

    results = []
    merged_ids = {r.session_id for r in rows if r.merged}

    while unmerged:
        # Pick next mergeable
        next_row = None
        for row in unmerged:
            deps = [d.strip() for d in row.depends_on if d.strip() != "none"]
            if all(d in merged_ids for d in deps):
                next_row = row
                break

        if next_row is None:
            results.append({"status": "deadlocked", "remaining": [r.session_id for r in unmerged]})
            break

        if dry_run:
            results.append({"session": next_row.session_id, "action": "would_merge"})
            unmerged.remove(next_row)
            merged_ids.add(next_row.session_id)
            continue

        # Confirm unless auto
        if not auto:
            if not questionary.confirm(f"Merge {next_row.session_id} (PR #{next_row.pr})?").ask():
                results.append({"session": next_row.session_id, "action": "skipped"})
                unmerged.remove(next_row)
                continue

        # Run exit_check in worktree
        worktree = repo_root / "worktrees" / next_row.session_id.split("-", 1)[1]
        if next_row.exit_check and next_row.exit_check != "none":
            check_result = subprocess.run(
                next_row.exit_check, shell=True, cwd=worktree, capture_output=True, text=True
            )
            if check_result.returncode != 0:
                # Exit check failed
                results.append({
                    "session": next_row.session_id,
                    "action": "exit_check_failed",
                    "stdout": check_result.stdout,
                    "stderr": check_result.stderr,
                })
                if not auto:
                    if not questionary.confirm(f"Exit check FAILED. Merge anyway?").ask():
                        unmerged.remove(next_row)
                        continue

        # Merge
        subprocess.run(["git", "checkout", "main"], cwd=repo_root, check=True)
        subprocess.run(["git", "pull"], cwd=repo_root, check=True)
        merge_result = subprocess.run(
            ["git", "merge", "--no-ff", f"session/{next_row.session_id}"],
            cwd=repo_root, capture_output=True, text=True
        )
        if merge_result.returncode != 0:
            results.append({
                "session": next_row.session_id,
                "action": "merge_conflict",
                "stderr": merge_result.stderr,
            })
            subprocess.run(["git", "merge", "--abort"], cwd=repo_root)
            unmerged.remove(next_row)
            continue

        # Push
        subprocess.run(["git", "push", "origin", "main"], cwd=repo_root, check=True)

        # Archive session log
        archive_session(repo_root, next_row.session_id)

        # Tick the box in queue
        mark_merged(queue_path, next_row.session_id)

        # Record decision
        write_decision_log(repo_root, next_row, "merged")

        merged_ids.add(next_row.session_id)
        unmerged.remove(next_row)
        results.append({"session": next_row.session_id, "action": "merged"})

    return results


def archive_session(repo_root: Path, session_id: str):
    log = repo_root / "logs" / "sessions" / f"{session_id}.md"
    archive_dir = repo_root / "logs" / "archive" / datetime.now().strftime("%Y-%m-%d")
    archive_dir.mkdir(parents=True, exist_ok=True)
    log.rename(archive_dir / f"{session_id}.md")


def mark_merged(queue_path: Path, session_id: str):
    content = queue_path.read_text()
    # Replace `- [ ]` with `- [x]` on the line containing session_id
    lines = content.splitlines()
    for i, line in enumerate(lines):
        if f" {session_id} " in line and line.startswith("- [ ]"):
            lines[i] = line.replace("- [ ]", "- [x]", 1)
            break
    queue_path.write_text("\n".join(lines) + "\n")


def write_decision_log(repo_root: Path, row: MergeQueueRow, action: str):
    decisions_dir = repo_root / "logs" / "decisions"
    decisions_dir.mkdir(exist_ok=True)
    ts = datetime.now().strftime("%Y%m%d-%H%M%S")
    log = decisions_dir / f"merge-{ts}-{row.session_id}.md"
    log.write_text(f"""---
timestamp: {datetime.now().isoformat()}
session: {row.session_id}
pr: {row.pr}
action: {action}
---

Merged {row.session_id} (PR #{row.pr}) into main.
Paths: {', '.join(row.touches_paths)}
Exit check: {row.exit_check}
""")
```

### CLI

```python
@app.command()
def merge(
    dry_run: bool = typer.Option(False, "--dry-run"),
    auto: bool = typer.Option(False, "--auto", "-a", help="No prompts"),
    reorder: bool = typer.Option(False, "--reorder", help="Interactive reorder"),
):
    """Reconcile the merge queue — land PRs in dependency order."""
    from .merger import reconcile

    if reorder:
        interactive_reorder()
        return

    results = reconcile(Path.cwd(), dry_run=dry_run, auto=auto)

    for r in results:
        if r["action"] == "merged":
            console.print(f"✓ [green]{r['session']}[/] merged")
        elif r["action"] == "exit_check_failed":
            console.print(f"✗ [red]{r['session']}[/] exit check failed:\n{r['stderr']}")
        elif r["action"] == "merge_conflict":
            console.print(f"✗ [red]{r['session']}[/] merge conflict — aborted")
        elif r["action"] == "would_merge":
            console.print(f"⊙ [yellow]{r['session']}[/] would merge")
```

## Edge cases

- **Queue row for non-existent branch** — log warning, skip, don't fail
- **main out of sync with origin** — `git pull` before merge; fail if pull has conflicts
- **PR already merged on GitHub but queue not ticked** — detect via `gh pr view`, tick locally
- **Hook failure (pre-commit, pre-push)** — propagate error, do NOT `--no-verify`
- **Worktree already removed** — exit check can't run; prompt human or `--force-merge`

- Don't force-push main
