# p3 — `swarm dispatch --local`

> **Why local first:** free via Max 20x, proven today, unblocks end-to-end validation of the CLI against ACT before we add cloud complexity.

---

## Goal

Implement the `swarm dispatch --local <task_id>` subcommand:
1. Read the composed prompt from `prompts/<task_id>.md`
2. Create git worktree + branch + session log (ports today's `session_init.sh`)
3. Launch Claude Code in a named tmux pane with the prompt
4. Register the session in `logs/sessions/` with status `INIT`
5. Return the tmux pane name and Langfuse trace URL

Multi-task variant: `swarm dispatch --all-ready` loops over ready tasks respecting concurrency limits.

---

## Success criteria

- [ ] `swarm dispatch --local T-X` reproduces today's s1-s4 fleet pattern
- [ ] Dispatcher interface is abstract (local now, cloud later in p6 without rewrites)
- [ ] Worktree created at `worktrees/<slug>/` on branch `session/<slug>`
- [ ] Session log at `logs/sessions/<slug>.md` with full frontmatter per p0 contract
- [ ] tmux pane named `swarm-<sid>` running `claude` with the prompt piped in
- [ ] Heartbeat env vars set: `CLAUDE_CODE_ENABLE_TELEMETRY=1`, `OTEL_EXPORTER_OTLP_ENDPOINT`, `OTEL_RESOURCE_ATTRIBUTES=session.id=<sid>,task.id=<tid>`
- [ ] Pre-flight checks: tmux installed, claude CLI on PATH, worktree path free, agent exists
- [ ] `--all-ready` respects config `max_concurrent_sessions` (default 4)
- [ ] Dry-run `--dry-run` prints what it would do without executing
- [ ] Idempotent: re-running `dispatch T-X` when session exists prints "already dispatched, see session s-X"

```python
# agentswarm/dispatchers/base.py
from abc import ABC, abstractmethod
from pathlib import Path
from ..schemas import TaskFrontmatter, SessionLogFrontmatter

class Dispatcher(ABC):
    """Common interface for local + cloud dispatchers."""

    @abstractmethod
    def dispatch(self, task: TaskFrontmatter, prompt: str, prompts_dir: Path) -> SessionHandle:
        """Launch a session, return a handle."""

    @abstractmethod
    def status(self, handle: SessionHandle) -> str:
        """Return current status (INIT/COOKING/etc.)."""

    @abstractmethod
    def tail(self, handle: SessionHandle, lines: int = 20) -> str:
        """Return last N lines of session output."""

    @abstractmethod
    def shutdown(self, handle: SessionHandle) -> None:
        """Gracefully stop the session."""


# agentswarm/dispatchers/local.py
import subprocess, os
from pathlib import Path
from .base import Dispatcher
from ..schemas import TaskFrontmatter, SessionLogFrontmatter
from ..state import write_session_log

class LocalDispatcher(Dispatcher):
    """Dispatch via git worktree + tmux + claude CLI, authenticated by Max 20x OAuth."""

    def __init__(self, config):
        self.config = config
        self.tmux_prefix = config.get("tmux_session_prefix", "swarm-")

    def dispatch(self, task, prompt, prompts_dir):
        # 1. Pre-flight
        self._check_tmux()
        self._check_claude()

        slug = task.slug
        worktree = Path(self.config["worktree_dir"]) / slug
        branch = f"session/{slug}"
        session_log = Path("logs/sessions") / f"{slug}.md"
        pane_name = f"{self.tmux_prefix}{slug}"

        if worktree.exists():
            raise AlreadyDispatchedError(f"worktree {worktree} exists")

        # 2. Create worktree + branch
        subprocess.run(["git", "worktree", "add", "-b", branch, str(worktree)], check=True)

        # 3. Write session log
        fm = SessionLogFrontmatter(
            session_id=slug,
            task_id=task.id,
            slug=slug,
            branch=branch,
            worktree=str(worktree),
            container=pane_name,
            mode="local",
            forked_from_sha=git_head_sha(),
            created=now_iso(),
            status="INIT",
            last_heartbeat=now_iso(),
            last_note="scaffolded by swarm dispatch",
            primary_agent=task.agent,
            model=task.prefer_model or self.config.get("default_model", "sonnet-4.6"),
            touches_paths=task.touches_paths,
            exit_check=task.exit_check or "",
        )
        write_session_log(session_log, fm, body_from_task(task))

        # 4. Launch tmux pane
        env = os.environ.copy()
        env["CLAUDE_CODE_ENABLE_TELEMETRY"] = "1"
        env["CLAUDE_CODE_ENHANCED_TELEMETRY_BETA"] = "1"
        env["OTEL_EXPORTER_OTLP_ENDPOINT"] = self.config.get("otel_endpoint", "http://localhost:3000/api/public/otel")
        env["OTEL_RESOURCE_ATTRIBUTES"] = f"session.id={slug},task.id={task.id},agent={task.agent}"

        # Write prompt to a temp file inside worktree for `claude < prompt.md`
        prompt_file = worktree / ".swarm-prompt.md"
        prompt_file.write_text(prompt)

        subprocess.run([
            "tmux", "new-session", "-d", "-s", pane_name,
            "-c", str(worktree),
            f"claude < {prompt_file}"
        ], check=True, env=env)

        return SessionHandle(
            session_id=slug,
            tmux_pane=pane_name,
            worktree=worktree,
            branch=branch,
            log_path=session_log,
        )

    def status(self, handle):
        # Parse status: frontmatter of log file
        fm = parse_session_log(handle.log_path)
        return fm.status

    def tail(self, handle, lines=20):
        result = subprocess.run(
            ["tmux", "capture-pane", "-t", handle.tmux_pane, "-p", "-S", f"-{lines}"],
            capture_output=True, text=True
        )
        return result.stdout

    def shutdown(self, handle):
        subprocess.run(["tmux", "kill-session", "-t", handle.tmux_pane])
```

### CLI

```python
@app.command()
def dispatch(
    task_id: str = typer.Argument(None),
    all_ready: bool = typer.Option(False, "--all-ready"),
    mode: str = typer.Option(None, "--mode", help="local | cloud (default: config)"),
    dry_run: bool = typer.Option(False, "--dry-run"),
    yes: bool = typer.Option(False, "--yes", "-y"),
):
    """Dispatch task(s) to Claude Code session(s)."""
    from .config import load_config
    from .scanner import scan
    from .composer import compose
    from .dispatchers.local import LocalDispatcher

    config = load_config(Path.cwd())
    mode = mode or config["dispatcher"]["default"]

    # Select dispatcher
    if mode == "local":
        dispatcher = LocalDispatcher(config["dispatcher"]["local"])
    elif mode == "cloud":
        from .dispatchers.cloud import CloudDispatcher  # p6
        dispatcher = CloudDispatcher(config["dispatcher"]["cloud"])
    else:
        raise typer.BadParameter(f"unknown mode {mode}")

    # Select tasks
    result = scan(Path.cwd())
    if all_ready:
        targets = result.ready
    else:
        targets = [t for t in result.ready if t.id == task_id]

    # Preview + confirm
    console.print(f"\n[bold]About to dispatch {len(targets)} task(s) in [cyan]{mode}[/] mode:[/]")
    for t in targets:
        console.print(f"  • {t.id} — {t.title}  ({t.agent}, ${t.budget_usd})")
    if not yes and not questionary.confirm("Proceed?").ask():
        raise typer.Abort()

    # Dispatch
    from rich.progress import Progress
    handles = []
    with Progress() as progress:
        for t in progress.track(targets, description="Dispatching..."):
            if dry_run:
                console.print(f"[yellow]DRY[/] would dispatch {t.id}")
                continue
            prompt = compose(t, Path.cwd(), [x for x in result.ready if x.id != t.id])
            handle = dispatcher.dispatch(t, prompt, Path("prompts"))
            handles.append(handle)
            console.print(f"✓ [green]{handle.session_id}[/] at tmux://{handle.tmux_pane}")

    console.print(f"\n[bold green]Dispatched {len(handles)} session(s).[/]")
    console.print(f"Follow with: [cyan]swarm watch[/]")
```
